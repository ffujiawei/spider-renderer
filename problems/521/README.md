```python
import re
import execjs
from renderer.utils import retry_get

req_url = 'http://jn.lssggzy.com/jnweb/jyxx/071005/071005003'
response = retry_get(req_url)

# 利用正则提取数据
code = re.findall(r'''setTimeout\("(\w+)\((\d+)\)", 200\); (function.+;})''', response.text)[0]
source = code[2].replace('eval("qo=eval;qo(po);")', 'return po')
ctx = execjs.compile(source)
res = ctx.call(code[0], code[1])
cookie = re.findall(r'''cookie='(\S+)=(\S+);''', res)[0]

# 设置
cookies = response.cookies.get_dict()
cookies[cookie[0]] = cookie[1]

# 再次请求
resp= retry_get(req_url, cookies=cookies)
```
